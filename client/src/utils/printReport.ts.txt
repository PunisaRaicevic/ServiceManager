import html2pdf from "html2pdf.js";

export function printReport(elementId: string, fileName = "report.pdf") {
  const element = document.getElementById(elementId);
  if (!element) {
    console.error(`Element with ID "${elementId}" not found`);
    return;
  }

  const options = {
    margin: 0.5,
    filename: fileName,
    image: { type: "jpeg", quality: 0.98 },
    html2canvas: { scale: 2 },
    jsPDF: { unit: "in", format: "a4", orientation: "portrait" },
  };

  // ✅ Radi i u browseru i u Android aplikaciji
  html2pdf()
    .set(options)
    .from(element)
    .outputPdf('bloburl')
    .then((pdfUrl: string) => {
      window.open(pdfUrl, '_blank'); // otvori PDF u pregledniku
    })
    .catch((err: any) => {
      console.error("Error generating PDF:", err);
    });
}
